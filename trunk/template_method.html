1. <?php
  2. /**
  3.  * TemplateMethod pattern
  4.  * 
  5.  * Executes this script and see the result
  6.  * matching the data provided at the bottom.
  7.  * 
  8.  * @author Jean-Lou Dupont
  9.  * @see    http://php-design-patterns-examples.googlecode.com/
  10.  */
  11.  
  12. abstract class template_method {
  13.  
  14. 	/**
  15. 	 * Simple state variable
  16. 	 */
  17. 	var $state;
  18.  
  19. 	/**
  20. 	 * Constructor
  21. 	 */
  22. 	public function __construct() {
  23.  
  24. 		$this->state = 0;
  25. 	}
  26. 	/**
  27. 	 * Controller method
  28. 	 */
  29. 	public function processEvent( $event ) {
  30.  
  31. 		$this->doFirstStep( $event );
  32. 		$this->doSecondStep( $event );
  33.  
  34. 	}
  35.  
  36. 	/**
  37. 	 * Abstract Method 'doFirstStep'
  38. 	 */
  39. 	abstract public function doFirstStep( &$event );
  40.  
  41. 	/**
  42. 	 * Abstract Method 'doSecondStep'
  43. 	 */
  44. 	abstract public function doSecondStep( &$event );
  45.  
  46. }
  47.  
  48. class CustomLogic
  49. 	extends template_method {
  50.  
  51. 	public function doFirstStep( &$event ) {
  52.  
  53. 		echo __METHOD__.": state: ".$this->state." event: $event\n";
  54. 		$this->state++;
  55. 	}
  56.  
  57. 	public function doSecondStep( &$event ) {
  58. 		echo __METHOD__.": state: ".$this->state." event: $event\n";
  59. 		$this->state++;
  60. 	}
  61.  
  62. }
  63.  
  64. /*****************************************************************
  65.  * 						EXAMPLE                                  *
  66.  *****************************************************************/
  67.  
  68. $cl = new CustomLogic;
  69.  
  70. $cl->processEvent( 'event1' );
  71. $cl->processEvent( 'event2' );
  72.  
  73. __halt_compiler();
  74.  
  75. /*****************************************************************
  76.  * 						RESULTS                                  *
  77.  *****************************************************************/
  78.  
  79. CustomLogic::doFirstStep: state: 0 event: event1
  80. CustomLogic::doSecondStep: state: 1 event: event1
  81. CustomLogic::doFirstStep: state: 2 event: event2
  82. CustomLogic::doSecondStep: state: 3 event: event2
