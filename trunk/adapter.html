1. <?php
  2. /**
  3.  * Adapter pattern
  4.  *
  5.  * Executes this script and see the result
  6.  * matching the data provided at the bottom.
  7.  *  
  8.  * @author Jean-Lou Dupont
  9.  * PROJECT SITE: //http://php-design-patterns-examples.googlecode.com/
  10.  */
  11.  
  12. class Server {
  13.  
  14. 	public function method( $p2, $p1 ) {
  15.  
  16. 		echo "Parameter1: $p1 , Parameter2: $p2 \n";
  17. 	}
  18. }
  19.  
  20. class Adaptor {
  21.  
  22. 	var $server;
  23.  
  24. 	public function __construct( &$server ) {
  25.  
  26. 		$this->server = $server;
  27. 	}
  28.  
  29. 	public function method( $p1, $p2 ) {
  30.  
  31. 		// the server just requires the two parameters
  32. 		// to be presented in reverse-order
  33. 		return $this->server->method( $p2, $p1 );
  34. 	}
  35. }
  36.  
  37. class Client {
  38.  
  39. 	var $server;
  40.  
  41. 	public function __construct( &$server ) {
  42.  
  43. 		$this->server = $server;
  44. 	}
  45.  
  46. 	public function execute( ) {
  47.  
  48. 		return $this->server->method( 'param1', 'param2' );
  49. 	}
  50.  
  51. }
  52.  
  53. /*****************************************************************
  54.  * 						EXAMPLE                                  *
  55.  *****************************************************************/
  56.  
  57. $server = new Server;
  58.  
  59. $adaptor = new Adaptor( $server );
  60.  
  61. $client = new Client( $adaptor );
  62.  
  63. $client->execute();
  64.  
  65. __halt_compiler();
  66.  
  67. /*****************************************************************
  68.  * 						RESULTS                                  *
  69.  *****************************************************************/
  70.  
  71. Parameter1: param1 , Parameter2: param2
