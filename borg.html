1. <?php
  2. /**
  3.  * Borg pattern
  4.  *  Variation on the Singleton pattern
  5.  * 
  6.  * Executes this script and see the result
  7.  * matching the data provided at the bottom.
  8.  * 
  9.  * @author Jean-Lou Dupont
  10.  * PROJECT SITE: //http://php-design-patterns-examples.googlecode.com/
  11.  */
  12.  
  13. class borg {
  14.  
  15. 	/**
  16. 	 * Shared Data
  17. 	 */
  18. 	private static $vars = array();
  19.  
  20. 	/**
  21. 	 * Default GETTER
  22. 	 */
  23. 	public function __get( $key ) {
  24.  
  25. 		if ( !isset( self::$vars[ $key ] ) )
  26. 			throw new Exception( "invalid key" );
  27.  
  28. 		return self::$vars[ $key ];
  29. 	}
  30. 	/**
  31. 	 * Default SETTER
  32. 	 */
  33. 	public function __set( $key, $value ) {
  34.  
  35. 		self::$vars[ $key ] = $value;
  36.  
  37. 	}
  38. }
  39.  
  40. class Test
  41. 	extends borg {
  42.  
  43. 	var $instanceData = null;
  44.  
  45. 	public function __construct( $name ) {
  46. 		$this->instanceData = $name;
  47. 	}
  48.  
  49. 	public function echoName() {
  50. 		echo "Class: ".__CLASS__.", instance name= ".$this->instanceData . "\n";
  51. 	}
  52. }
  53.  
  54. /*****************************************************************
  55.  * 						EXAMPLE                                  *
  56.  *****************************************************************/
  57.  
  58. $a = new Test( "instance A" );
  59. $b = new Test( "instance B" );
  60.  
  61. $a->sharedState = "Hi!\n";
  62.  
  63. echo $b->sharedState;
  64.  
  65. $a->echoName();
  66. $b->echoName();
  67.  
  68. __halt_compiler();
  69.  
  70. /*****************************************************************
  71.  * 						RESULTS                                  *
  72.  *****************************************************************/
  73.  
  74. Hi!
  75. Class: Test, instance name= instance A
  76. Class: Test, instance name= instance B
