1. <?php
  2. /**
  3.  * Factory pattern
  4.  *
  5.  * Executes this script and see the result
  6.  * matching the data provided at the bottom.
  7.  *  
  8.  * @author Jean-Lou Dupont
  9.  * PROJECT SITE: //http://php-design-patterns-examples.googlecode.com/
  10.  */
  11.  
  12. class Base {
  13.  
  14. 	/**
  15. 	 * Class map for the factory
  16. 	 * @access private
  17. 	 */
  18. 	private static $classMap = array();
  19.  
  20.  
  21. 	/*================================================================
  22. 	 					PUBLIC INTERFACE
  23. 	 ================================================================*/
  24.  
  25. 	/**
  26. 	 * Maps a class to another one
  27. 	 * Used along with the factory interface.
  28. 	 * 
  29. 	 * @param $oClasse string original class
  30. 	 * @param $rClasse string replacement class
  31. 	 * @return void
  32. 	 */
  33. 	public static function mapClass( $oClasse, $rClasse ) {
  34.  
  35. 		self::$classMap[ $oClasse ] = $rClasse;
  36. 	}
  37. 	/**
  38. 	 * factory
  39. 	 * 
  40. 	 * @param $className name of class to instantiate/get (singleton)
  41. 	 * @param mixed optional parameters
  42. 	 * @return $object
  43. 	 */
  44. 	public static function factory( $className /*optional params*/ ) {
  45.  
  46. 		$args = func_get_args();
  47. 		array_shift( $args );	#get rid of $classname
  48.  
  49. 		$classe = self::map( $className );
  50.  
  51. 		return self::buildObject( $classe, $args );
  52. 	}
  53.  
  54. 	/*================================================================
  55. 	 					PRIVATE INTERFACE
  56. 	 ================================================================*/
  57.  
  58. 	/**
  59. 	 * Maps class-name
  60. 	 * @param string $className
  61. 	 * @return string $out_class
  62. 	 */
  63. 	private static function map( &$className ) {
  64.  
  65. 		// check our replacement map
  66. 		if ( isset( self::$classMap[$className] )) {
  67.  
  68. 			$classe = self::$classMap[$className];
  69.  
  70. 			// autoloads & verifies existence
  71. 			if ( !class_exists( $classe, true ))
  72. 				throw new Exception( "unable to load class ($className) remapped to ($classe)" );
  73.  
  74. 			return $classe;
  75. 		}
  76.  
  77. 		// default i.e. no remapping
  78. 		if ( !class_exists( $className, true ))
  79. 			throw new Exception( "unable to load class $className" );
  80.  
  81. 		// if no mapping is required
  82. 		return $className;
  83. 	}
  84.  
  85. 	/**
  86. 	 * Builds an object instance of the specified class
  87. 	 * and passes properly a variable argument list
  88. 	 * 
  89. 	 * @param $classe
  90. 	 * @param $args mixed
  91. 	 * @return $object
  92. 	 * @throws aop_exception when too many arguments are passed in array $args
  93. 	 */
  94. 	private static function buildObject( &$classe, Array &$args ) {
  95.  
  96. 		$count = count( $args );
  97.  
  98. 		switch( $count ) {
  99. 		case 0:
  100. 			return new $classe;
  101. 		case 1:
  102. 			return new $classe( $args[0] );
  103. 		case 2:
  104. 			return new $classe( $args[0], $args[1] );
  105. 		case 3:
  106. 			return new $classe( $args[0], $args[1], $args[2] );
  107. 		case 4:
  108. 			return new $classe( $args[0], $args[1], $args[2], $args[3] );
  109. 		default:
  110. 			throw new Exception( "unsupported number of arguments whilst creating object in factory" );
  111. 		}
  112. 	}
  113.  
  114.  
  115. }//end class
  116.  
  117.  
  118. /*****************************************************************
  119.  * 						EXAMPLE                                  *
  120.  *****************************************************************/
  121.  
  122. class Test1 {}
  123.  
  124. class Test2 {}
  125.  
  126. class Test3 {
  127.  
  128. 	var $p1;
  129. 	var $p2;
  130.  
  131. 	public function __construct( $required_param1, $required_param2 ) {
  132. 		$this->p1 = $required_param1;
  133. 		$this->p2 = $required_param2;		
  134. 	}
  135. }
  136.  
  137. class Test4 {
  138.  
  139. 	var $p1;
  140. 	var $p2;
  141. 	var $p3;
  142.  
  143. 	public function __construct( $required_param1, $required_param2, $optional_param = null ) {
  144. 		$this->p1 = $required_param1;
  145. 		$this->p2 = $required_param2;		
  146. 		$this->p3 = $optional_param;
  147. 	}
  148. }
  149.  
  150. // simple case
  151. $o1 = Base::factory( 'Test1' );
  152. echo "o1: class=  ".get_class( $o1 )."\n";
  153.  
  154. // remapping case
  155. Base::mapClass( 'Test1', 'Test2' );
  156. $o2 = Base::factory( 'Test1' );
  157. echo "o2: class=  ".get_class( $o2 )."\n";
  158.  
  159. // remapping & creation with parameters case
  160. Base::mapClass( 'Test3', 'Test4' );
  161. $o3 = Base::factory( 'Test3', 'req_param1', 'req_param2' );
  162. echo "o3: class=  ".get_class( $o3 )."\n";
  163. echo "o3: param1= ".$o3->p1." param2= ".$o3->p2."\n";
  164.  
  165.  
  166. __halt_compiler();
  167. /*****************************************************************
  168.  * 						RESULTS                                  *
  169.  *****************************************************************/
  170.  
  171. o1: class=  Test1
  172. o2: class=  Test2
  173. o3: class=  Test4
  174. o3: param1= req_param1 param2= req_param2
